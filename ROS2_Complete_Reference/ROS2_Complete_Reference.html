🔵 2. Nodes

A node is a single executable process that does one thing. Nodes communicate via topics, services, and actions. One package can have multiple nodes.

Python node — minimal

import rclpy
from rclpy.node import Node

class MyNode(Node):
    def __init__(self):
        # 'my_node' is the node name seen in ros2 node list
        super().__init__('my_node')
        self.get_logger().info('Node started!')

        # Timer — runs callback every 0.5 seconds
        self.timer = self.create_timer(0.5, self.timer_cb)

    def timer_cb(self):
        self.get_logger().info('Tick')

def main(args=None):
    rclpy.init(args=args)          # init ROS 2
    node = MyNode()
    rclpy.spin(node)               # keep alive, process callbacks
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
    main()

C++ node — minimal

#include "rclcpp/rclcpp.hpp"

class MyNode : public rclcpp::Node {
public:
    MyNode() : Node("my_node") {
        RCLCPP_INFO(get_logger(), "Node started!");
        timer_ = create_wall_timer(
            std::chrono::milliseconds(500),
            std::bind(&MyNode::timer_cb, this));
    }
private:
    void timer_cb() {
        RCLCPP_INFO(get_logger(), "Tick");
    }
    rclcpp::TimerBase::SharedPtr timer_;
};

int main(int argc, char** argv) {
    rclcpp::init(argc, argv);
    rclcpp::spin(std::make_shared<MyNode>());
    rclcpp::shutdown();
    return 0;
}

📡 3. Topics — Publisher & Subscriber

When to use: Continuous streaming data — sensor readings, images, velocity commands, pose updates. Fire-and-forget. No response expected.

Publisher (Python)

from std_msgs.msg import String
from geometry_msgs.msg import Twist

class MyPublisher(Node):
    def __init__(self):
        super().__init__('publisher_node')

        # Create publisher: topic name, msg type, queue size
        self.pub = self.create_publisher(
            Twist, '/cmd_vel', 10)

        self.timer = self.create_timer(0.1, self.publish_cb)

    def publish_cb(self):
        msg = Twist()
        msg.linear.x = 0.2   # forward
        msg.angular.z = 0.0  # no rotation
        self.pub.publish(msg)
        self.get_logger().info('Published velocity')

Subscriber (Python)

from sensor_msgs.msg import LaserScan
from nav_msgs.msg import OccupancyGrid

class MySubscriber(Node):
    def __init__(self):
        super().__init__('subscriber_node')

        # Subscribe: topic, msg type, callback, queue size
        self.sub = self.create_subscription(
            LaserScan,
            '/scan',
            self.scan_callback,
            10)

    def scan_callback(self, msg: LaserScan):
        # msg contains all fields of the message type
        min_range = min(r for r in msg.ranges
                        if r > msg.range_min)
        self.get_logger().info(
            f'Closest obstacle: {min_range:.2f}m')

🔧 4. Services — Request / Response

When to use: One-time operations where you need a result back. Examples: clear costmap, get map, set a parameter, trigger a camera. NOT for continuous data.

Service Server (Python)

from std_srvs.srv import SetBool
from example_interfaces.srv import AddTwoInts

class MyServer(Node):
    def __init__(self):
        super().__init__('my_server')
        self.srv = self.create_service(
            SetBool,          # service type
            'enable_motor',  # service name
            self.handle_req)  # callback

    def handle_req(self, request, response):
        # request.data is the bool sent by client
        if request.data:
            self.get_logger().info('Motor ON')
        response.success = True
        response.message = 'Done'
        return response  # MUST return response

Service Client (Python)

from std_srvs.srv import SetBool

class MyClient(Node):
    def __init__(self):
        super().__init__('my_client')
        self.cli = self.create_client(
            SetBool, 'enable_motor')

        # Wait for service to be available
        while not self.cli.wait_for_service(timeout_sec=1.0):
            self.get_logger().warn('Waiting for service...')

    def call_service(self, enable: bool):
        req = SetBool.Request()
        req.data = enable
        future = self.cli.call_async(req)
        # Spin until future completes
        rclpy.spin_until_future_complete(self, future)
        return future.result()

🎯 5. Actions — Long Running Goals with Feedback

When to use: Long tasks where you need progress updates AND can cancel. Navigation goals, robot arm movements, multi-step operations. This is how Nav2 NavigateToPose works.

Action Server (Python)

import rclpy, time
from rclpy.action import ActionServer
from action_tutorials_interfaces.action import Fibonacci

class MyActionServer(Node):
    def __init__(self):
        super().__init__('action_server')
        self._server = ActionServer(
            self,
            Fibonacci,            # action type
            'compute_fibonacci',  # action name
            self.execute_cb)      # execution callback

    def execute_cb(self, goal_handle):
        self.get_logger().info('Goal received!')
        result = Fibonacci.Result()
        feedback = Fibonacci.Feedback()

        sequence = [0, 1]
        for i in range(1, goal_handle.request.order):
            sequence.append(sequence[i] + sequence[i-1])
            feedback.partial_sequence = sequence
            goal_handle.publish_feedback(feedback)  # send progress
            time.sleep(0.5)

        goal_handle.succeed()         # mark goal as done
        result.sequence = sequence
        return result

Action Client (Python)

from rclpy.action import ActionClient
from nav2_msgs.action import NavigateToPose
from geometry_msgs.msg import PoseStamped

class NavClient(Node):
    def __init__(self):
        super().__init__('nav_client')
        self.client = ActionClient(self, NavigateToPose, '/navigate_to_pose')

    def send_goal(self, x, y, yaw):
        goal = NavigateToPose.Goal()
        goal.pose.header.frame_id = 'map'
        goal.pose.pose.position.x = x
        goal.pose.pose.position.y = y
        # yaw to quaternion
        import math
        goal.pose.pose.orientation.z = math.sin(yaw/2)
        goal.pose.pose.orientation.w = math.cos(yaw/2)

        self.client.wait_for_server()
        future = self.client.send_goal_async(
            goal,
            feedback_callback=self.feedback_cb)
        future.add_done_callback(self.goal_response_cb)

    def feedback_cb(self, feedback_msg):
        dist = feedback_msg.feedback.distance_remaining
        self.get_logger().info(f'Distance remaining: {dist:.2f}m')

    def goal_response_cb(self, future):
        handle = future.result()
        if not handle.accepted:
            self.get_logger().error('Goal rejected!')
            return
        result_future = handle.get_result_async()
        result_future.add_done_callback(self.result_cb)

    def result_cb(self, future):
        result = future.result().result
        self.get_logger().info('Goal reached!')

from launch.substitutions import LaunchConfiguration

import os

def generate_launch_description():
    # Declare argument with default

        'use_sim_time', default_value='false')

    # Launch a node
    my_node = Node(
        package='my_package',
        executable='my_node',
        name='my_node',
        output='screen',
        parameters=[{
            'use_sim_time': LaunchConfiguration('use_sim_time'),
            'speed': 0.2,

        remappings=[('/cmd_vel', '/robot/cmd_vel')]

    pkg = get_package_share_directory('nav2_bringup')
    nav2 = IncludeLaunchDescription(
        os.path.join(pkg, 'launch', 'navigation_launch.py'))

    return LaunchDescription([use_sim, my_node, nav2])

🌐 8. TF2 — Transform System

TF2 tracks the coordinate frames of every part of the robot (base, sensors, map) and lets you transform points between them. Everything in Nav2 depends on TF2.

Key frames to know

FrameWhat it representsPublished by
mapFixed global map frame. Origin = map (0,0)map_server / AMCL
odomOdometry frame — drifts over timeRobot base driver
base_linkRobot body centreRobot URDF / driver
base_footprintRobot projected onto floorRobot URDF
laser_frameLIDAR sensor originLIDAR driver
TF Chainmap → odom → base_link → laser_frame. Each arrow is a transform. AMCL provides map→odom. The driver provides odom→base_link.

Broadcast a transform (Python)

from tf2_ros import TransformBroadcaster
from geometry_msgs.msg import TransformStamped

class TFBroadcaster(Node):
    def __init__(self):
        super().__init__('tf_broadcaster')
        self.br = TransformBroadcaster(self)
        self.timer = self.create_timer(0.1, self.broadcast)

    def broadcast(self):
        t = TransformStamped()
        t.header.stamp = self.get_clock().now().to_msg()
        t.header.frame_id = 'base_link'   # parent
        t.child_frame_id  = 'laser_frame' # child
        t.transform.translation.x = 0.04
        t.transform.rotation.w = 1.0      # no rotation
        self.br.sendTransform(t)

Listen to a transform (Python)

from tf2_ros import Buffer, TransformListener

class TFListener(Node):
    def __init__(self):
        super().__init__('tf_listener')
        self.tf_buffer = Buffer()
        self.tf_listener = TransformListener(
            self.tf_buffer, self)

    def get_robot_pose(self):
        try:
            t = self.tf_buffer.lookup_transform(
                'map',        # target frame
                'base_link',  # source frame
                rclpy.time.Time())
            x = t.transform.translation.x
            y = t.transform.translation.y
            return x, y
        except Exception as e:
            self.get_logger().warn(f'TF error: {e}')

AMCL — Localisation

AMCL uses particle filters to estimate robot position on the map using LIDAR scan matching. It REQUIRES an initial pose estimate before it works.

# Set initial pose from terminal
ros2 topic pub /initialpose geometry_msgs/PoseWithCovarianceStamped \
"{header: {frame_id: map}, pose: {pose: {position: {x: 0.0, y: 0.0}, orientation: {w: 1.0}}}}" --once

# Watch AMCL output
ros2 topic echo /amcl_pose

# Check particle cloud
ros2 topic echo /particle_cloud
Common mistakeIf the robot is not moving or navigation fails, first check if initial pose has been set. AMCL without initial pose = completely wrong localisation.

Sending a navigation goal (Python node)

import rclpy, math
from rclpy.node import Node
from rclpy.action import ActionClient
from nav2_msgs.action import NavigateToPose

class Navigator(Node):
    def __init__(self):
        super().__init__('navigator')
        self.ac = ActionClient(self, NavigateToPose, '/navigate_to_pose')

    def go_to(self, x, y, yaw=0.0):
        goal = NavigateToPose.Goal()
        goal.pose.header.frame_id = 'map'
        goal.pose.header.stamp = self.get_clock().now().to_msg()
        goal.pose.pose.position.x = x
        goal.pose.pose.position.y = y
        goal.pose.pose.orientation.z = math.sin(yaw/2)
        goal.pose.pose.orientation.w = math.cos(yaw/2)

        self.ac.wait_for_server()
        self.get_logger().info(f'Sending goal: ({x},{y})')
        future = self.ac.send_goal_async(goal, self.feedback_cb)
        future.add_done_callback(self.goal_cb)

    def feedback_cb(self, fb):
        d = fb.feedback.distance_remaining
        self.get_logger().info(f'Distance left: {d:.2f}m')

    def goal_cb(self, future):
        gh = future.result()
        gh.get_result_async().add_done_callback(self.result_cb)

    def result_cb(self, future):
        self.get_logger().info('GOAL REACHED!')

def main():
    rclpy.init()
    nav = Navigator()
    nav.go_to(1.5, -0.5, 1.57)
    rclpy.spin(nav)

DWB — understanding the critics

CriticWhat it penalisesEffect
BaseObstacleTrajectory passing through costmap lethal cellsCore collision avoidance — infinite cost = rejected
PathDistLateral distance from global planKeeps robot on the planned route
GoalDistDistance remaining to goal along pathPrefers trajectories that make progress
PathAlignHeading difference vs global plan directionKeeps robot pointing along the route
RotateToGoalHeading error vs goal orientation when closeRobot turns correctly before stopping
OscillationRepeated direction reversalsPrevents spinning back and forth

Node template — copy this for every new node

import rclpy
from rclpy.node import Node
from std_msgs.msg import String          # change as needed
from std_srvs.srv import SetBool          # change as needed

class MyNode(Node):
    def __init__(self):
        super().__init__('my_node')

        # Parameters
        self.declare_parameter('speed', 0.2)
        self.speed = self.get_parameter('speed').value

        # Publisher
        self.pub = self.create_publisher(String, '/output', 10)

        # Subscriber
        self.sub = self.create_subscription(String, '/input', self.input_cb, 10)

        # Service server
        self.srv = self.create_service(SetBool, '~/enable', self.srv_cb)

        # Timer
        self.timer = self.create_timer(0.5, self.timer_cb)

        self.get_logger().info('MyNode started')

    def timer_cb(self):
        msg = String()
        msg.data = 'hello'
        self.pub.publish(msg)

    def input_cb(self, msg):
        self.get_logger().info(f'Received: {msg.data}')

    def srv_cb(self, req, res):
        res.success = True
        res.message = 'OK'
        return res

def main(args=None):
    rclpy.init(args=args)
    node = MyNode()
    try:
        rclpy.spin(node)
    except KeyboardInterrupt:
        pass
    finally:
        node.destroy_node()
        rclpy.shutdown()

if __name__ == '__main__':
    main()

Node that counts how many times it's been alive
Warm-up
🎯 Goal

Create a node called uptime_counter. Every 2 seconds it logs how many seconds it has been running. Format: [INFO] Running for 6 seconds

📝 Your task
  • Create a new Python file
  • Make a Node class
  • Use a timer to count elapsed seconds
  • Log the count each tick
Think before coding: What variable do you need? How do you calculate elapsed time from ticks? What timer interval makes the count work correctly? 
import rclpy
from rclpy.node import Node

class UptimeCounter(Node):
    def __init__(self):
        super().__init__('uptime_counter')
        self.seconds = 0
        # Timer fires every 1 second — each tick = 1 second elapsed
        self.timer = self.create_timer(1.0, self.tick)

    def tick(self):
        self.seconds += 1
        if self.seconds % 2 == 0:  # only log every 2 seconds
            self.get_logger().info(f'Running for {self.seconds} seconds')

def main(args=None):
    rclpy.init(args=args)
    node = UptimeCounter()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
    main()
Run it: python3 uptime_counter.py — You should see the count every 2 seconds.

Temperature sensor simulator — publish fake sensor data
Core
🎯 Goal

Simulate a temperature sensor. Publish a Float64 value to /temperature every second. Start at 25.0°C and add a small random fluctuation (±0.5°C) each tick. Log the published value.

Think: Which message type do you import? What does random fluctuation mean in code? How do you clamp a value between a range?
💡 Hints
  • Import: from std_msgs.msg import Float64
  • Import: import random
  • random.uniform(-0.5, 0.5) gives a random float between -0.5 and 0.5
import rclpy, random
from rclpy.node import Node
from std_msgs.msg import Float64

class TempSensor(Node):
    def __init__(self):
        super().__init__('temp_sensor')
        self.pub = self.create_publisher(Float64, '/temperature', 10)
        self.temp = 25.0
        self.timer = self.create_timer(1.0, self.publish_temp)

    def publish_temp(self):
        self.temp += random.uniform(-0.5, 0.5)
        self.temp = max(20.0, min(30.0, self.temp))  # clamp 20-30°C
        msg = Float64()
        msg.data = self.temp
        self.pub.publish(msg)
        self.get_logger().info(f'Temperature: {self.temp:.2f}°C')

def main(args=None):
    rclpy.init(args=args)
    rclpy.spin(TempSensor())
    rclpy.shutdown()

if __name__ == '__main__':
    main()

Temperature monitor — subscribe and classify readings
Core
🎯 Goal

Subscribe to /temperature (from Ex 2). For each reading, classify and log it:

  • Below 22°C → COLD
  • 22–27°C → NORMAL
  • Above 27°C → HOT — check cooling system

Also track and log the running average temperature.

Think: How do you calculate a running average without storing all values? (hint: keep a count and a total) 
import rclpy
from rclpy.node import Node
from std_msgs.msg import Float64

class TempMonitor(Node):
    def __init__(self):
        super().__init__('temp_monitor')
        self.sub = self.create_subscription(
            Float64, '/temperature', self.on_temp, 10)
        self.total = 0.0
        self.count = 0

    def on_temp(self, msg: Float64):
        t = msg.data
        self.total += t
        self.count += 1
        avg = self.total / self.count

        if t < 22.0:
            status = 'COLD'
        elif t > 27.0:
            status = 'HOT — check cooling system'
        else:
            status = 'NORMAL'

        self.get_logger().info(
            f'Temp: {t:.2f}°C | Status: {status} | Avg: {avg:.2f}°C')

def main(args=None):
    rclpy.init(args=args)
    rclpy.spin(TempMonitor())
    rclpy.shutdown()

if __name__ == '__main__':
    main()

Reset service — add a /reset_avg service to the monitor
Core
🎯 Goal

Extend the TempMonitor from Ex 3. Add a service /reset_avg using std_srvs/Empty that resets the total and count back to zero. Log "Average reset!" when called.

💡 Key import
from std_srvs.srv import Empty

Empty service has no request or response fields — you just call it and the callback fires.

Think: The service callback signature for Empty is: def handle(self, request, response) — same as always, just the request and response have no fields. You still must return response. 
import rclpy
from rclpy.node import Node
from std_msgs.msg import Float64
from std_srvs.srv import Empty

class TempMonitor(Node):
    def __init__(self):
        super().__init__('temp_monitor')
        self.sub = self.create_subscription(
            Float64, '/temperature', self.on_temp, 10)
        self.srv = self.create_service(
            Empty, '/reset_avg', self.handle_reset)
        self.total = 0.0
        self.count = 0

    def on_temp(self, msg):
        t = msg.data
        self.total += t
        self.count += 1
        avg = self.total / self.count
        self.get_logger().info(f'Temp: {t:.2f}°C | Avg: {avg:.2f}°C')

    def handle_reset(self, request, response):
        self.total = 0.0
        self.count = 0
        self.get_logger().info('Average reset!')
        return response  # always return response!

def main(args=None):
    rclpy.init(args=args)
    rclpy.spin(TempMonitor())
    rclpy.shutdown()

if __name__ == '__main__':
    main()

Parameters — make the temperature thresholds configurable
Core
🎯 Goal

Extend TempMonitor so the HOT and COLD thresholds are ROS 2 parameters, not hardcoded values. Default: cold_threshold=22.0, hot_threshold=27.0. Change them at runtime without restarting the node.

Think: Parameters should be declared in __init__. But when should you READ them — at startup or on every callback? What's better for runtime changes? 
import rclpy
from rclpy.node import Node
from std_msgs.msg import Float64

class TempMonitor(Node):
    def __init__(self):
        super().__init__('temp_monitor')
        # Declare with defaults
        self.declare_parameter('cold_threshold', 22.0)
        self.declare_parameter('hot_threshold', 27.0)
        self.sub = self.create_subscription(
            Float64, '/temperature', self.on_temp, 10)

    def on_temp(self, msg):
        t = msg.data
        # Read params on every callback — picks up runtime changes
        cold = self.get_parameter('cold_threshold').value
        hot  = self.get_parameter('hot_threshold').value

        if t < cold:
            status = f'COLD (threshold {cold}°C)'
        elif t > hot:
            status = f'HOT (threshold {hot}°C)'
        else:
            status = 'NORMAL'

        self.get_logger().info(f'{t:.2f}°C — {status}')

def main(args=None):
    rclpy.init(args=args)
    rclpy.spin(TempMonitor())
    rclpy.shutdown()
Change at runtime: ros2 param set /temp_monitor hot_threshold 24.0 — now readings above 24° trigger HOT.

Motor speed controller — pub + sub + logic in one node
Challenge
🎯 Goal

Build a node that simulates a motor speed controller:

  • Subscribe to /target_speed (Float64) — the desired speed
  • Publish to /actual_speed (Float64) every 0.5s
  • The actual speed moves toward the target by 10% per tick (simulates motor inertia)
  • Log both target and actual speed each tick
Think — this is real control logic: If actual=0 and target=100, after one tick actual=10. After two ticks actual=19 (10% of 90 remaining). This is exponential approach — you've implemented it before in your turtle chaser (proportional control).
💡 Formula
actual = actual + (target - actual) * 0.1
import rclpy
from rclpy.node import Node
from std_msgs.msg import Float64

class MotorController(Node):
    def __init__(self):
        super().__init__('motor_controller')
        self.actual_speed = 0.0
        self.target_speed = 0.0

        self.sub = self.create_subscription(
            Float64, '/target_speed', self.on_target, 10)
        self.pub = self.create_publisher(
            Float64, '/actual_speed', 10)
        self.timer = self.create_timer(0.5, self.update)

    def on_target(self, msg):
        self.target_speed = msg.data

    def update(self):
        # Proportional approach — same math as your turtle chaser!
        self.actual_speed += (self.target_speed - self.actual_speed) * 0.1
        msg = Float64()
        msg.data = self.actual_speed
        self.pub.publish(msg)
        self.get_logger().info(
            f'Target: {self.target_speed:.1f} | Actual: {self.actual_speed:.2f}')

def main(args=None):
    rclpy.init(args=args)
    rclpy.spin(MotorController())
    rclpy.shutdown()
Send a target: ros2 topic pub /target_speed std_msgs/Float64 "{data: 100.0}" --once — watch actual_speed climb toward 100.

LIDAR obstacle detector — process real /scan data
Challenge
🎯 Goal

Subscribe to /scan (LaserScan). For each scan:

  • Filter out invalid readings (inf, nan, out of range)
  • Find the minimum distance across all valid readings
  • Find which direction it's in (convert index to angle in degrees)
  • If min distance < 0.5m → log OBSTACLE CLOSE at X.XXm, direction: YY°
  • Otherwise → log CLEAR — closest: X.XXm
  • Also publish a Bool to /obstacle_detected
Think: LaserScan has: msg.ranges (list of distances), msg.angle_min, msg.angle_increment, msg.range_min, msg.range_max. How do you get the angle for index i?
💡 Angle formula
angle_of_index_i = msg.angle_min + i * msg.angle_increment  # in radians
degrees = math.degrees(angle_of_index_i)
import rclpy, math
from rclpy.node import Node
from sensor_msgs.msg import LaserScan
from std_msgs.msg import Bool

class ObstacleDetector(Node):
    def __init__(self):
        super().__init__('obstacle_detector')
        self.sub = self.create_subscription(
            LaserScan, '/scan', self.on_scan, 10)
        self.pub = self.create_publisher(Bool, '/obstacle_detected', 10)

    def on_scan(self, msg: LaserScan):
        valid = []
        for i, r in enumerate(msg.ranges):
            if math.isfinite(r) and msg.range_min <= r <= msg.range_max:
                angle_deg = math.degrees(msg.angle_min + i * msg.angle_increment)
                valid.append((r, angle_deg))

        if not valid:
            return

        min_dist, min_angle = min(valid, key=lambda x: x[0])
        obstacle = min_dist < 0.5

        msg_out = Bool()
        msg_out.data = obstacle
        self.pub.publish(msg_out)

        if obstacle:
            self.get_logger().warn(
                f'OBSTACLE CLOSE at {min_dist:.2f}m, direction: {min_angle:.1f}°')
        else:
            self.get_logger().info(
                f'CLEAR — closest: {min_dist:.2f}m at {min_angle:.1f}°')

def main(args=None):
    rclpy.init(args=args)
    rclpy.spin(ObstacleDetector())
    rclpy.shutdown()
This is real code used in industry. Obstacle detection from LIDAR is exactly what My Equation will want you to write.

Velocity governor — stop robot if obstacle too close
Challenge
🎯 Goal

Build a safety node that sits between a velocity source and the robot:

  • Subscribe to /cmd_vel_raw (Twist) — the unsafe commands
  • Subscribe to /obstacle_detected (Bool) — from Ex 7
  • Publish to /cmd_vel (Twist) — what actually goes to the robot
  • If obstacle detected → publish zero velocity regardless of input
  • Otherwise → pass through the raw command unchanged
  • Log when you block a command vs pass it through
Think: This node has TWO subscribers. Each has its own callback. They share state (self.obstacle). The pub/sub cycle: raw_cmd → this node → /cmd_vel. This pattern is called a "safety filter" and is used in every real robot. 
import rclpy
from rclpy.node import Node
from geometry_msgs.msg import Twist
from std_msgs.msg import Bool

class VelocityGovernor(Node):
    def __init__(self):
        super().__init__('velocity_governor')
        self.obstacle = False

        self.sub_vel = self.create_subscription(
            Twist, '/cmd_vel_raw', self.on_vel, 10)
        self.sub_obs = self.create_subscription(
            Bool, '/obstacle_detected', self.on_obstacle, 10)
        self.pub = self.create_publisher(Twist, '/cmd_vel', 10)

    def on_obstacle(self, msg: Bool):
        self.obstacle = msg.data  # just update the flag

    def on_vel(self, msg: Twist):
        if self.obstacle:
            self.get_logger().warn('BLOCKED — obstacle too close, zeroing velocity')
            self.pub.publish(Twist())  # Twist() = all zeros by default
        else:
            self.get_logger().info(f'PASS — vx={msg.linear.x:.2f}')
            self.pub.publish(msg)

def main(args=None):
    rclpy.init(args=args)
    rclpy.spin(VelocityGovernor())
    rclpy.shutdown()

## NODE 1: machine_sensor.py
import rclpy, random
from rclpy.node import Node
from std_msgs.msg import Float64

class MachineSensor(Node):
    def __init__(self):
        super().__init__('machine_sensor')
        self.temp_pub  = self.create_publisher(Float64, '/machine/temp', 10)
        self.vib_pub   = self.create_publisher(Float64, '/machine/vibration', 10)
        self.rpm_pub   = self.create_publisher(Float64, '/machine/rpm', 10)
        self.temp = 60.0;  self.vib = 1.5;  self.rpm = 1200.0
        self.create_timer(1.0, self.publish_all)

    def publish_all(self):
        self.temp += random.uniform(-1, 2)   # drifts upward
        self.vib  += random.uniform(-0.2, 0.3)
        self.rpm  += random.uniform(-20, 10)
        for pub, val in [(self.temp_pub, self.temp),
                          (self.vib_pub,  self.vib),
                          (self.rpm_pub,  self.rpm)]:
            msg = Float64(); msg.data = val; pub.publish(msg)

## NODE 2: status_aggregator.py
class StatusAggregator(Node):
    def __init__(self):
        super().__init__('status_aggregator')
        self.temp = self.vib = self.rpm = None
        self.create_subscription(Float64, '/machine/temp',
                                  lambda m: setattr(self, 'temp', m.data), 10)
        self.create_subscription(Float64, '/machine/vibration',
                                  lambda m: setattr(self, 'vib', m.data), 10)
        self.create_subscription(Float64, '/machine/rpm',
                                  lambda m: setattr(self, 'rpm', m.data), 10)
        self.pub = self.create_publisher(
            __import__('std_msgs.msg', fromlist=['String']).String,
            '/machine/status', 10)
        self.create_timer(1.0, self.compute_status)

    def compute_status(self):
        if None in (self.temp, self.vib, self.rpm): return
        if self.temp > 80 or self.vib > 5.0 or self.rpm < 500:
            status = 'CRITICAL'
        elif self.temp > 65 or self.vib > 3.0 or self.rpm < 800:
            status = 'WARNING'
        else:
            status = 'OK'
        from std_msgs.msg import String
        msg = String(); msg.data = status; self.pub.publish(msg)

MINI PROJECT — cmd_vel publisher with keyboard-like logic
Mini Project
🎯 Goal

Build a node that reads from a parameter command (string) and publishes the matching Twist to /cmd_vel. Commands: "forward", "backward", "left", "right", "stop". Use parameters so you can change direction at runtime without restarting.

Think: This teaches you to control a robot purely with ros2 param set — no keyboard node needed. Used in testing and automation scripts. 
import rclpy
from rclpy.node import Node
from geometry_msgs.msg import Twist

COMMANDS = {
    'forward':  (0.2, 0.0),
    'backward': (-0.2, 0.0),
    'left':     (0.0, 0.5),
    'right':    (0.0, -0.5),
    'stop':     (0.0, 0.0),
}

class CmdVelDriver(Node):
    def __init__(self):
        super().__init__('cmd_vel_driver')
        self.declare_parameter('command', 'stop')
        self.pub = self.create_publisher(Twist, '/cmd_vel', 10)
        self.create_timer(0.1, self.drive)

    def drive(self):
        cmd = self.get_parameter('command').value
        lin, ang = COMMANDS.get(cmd, (0.0, 0.0))
        msg = Twist()
        msg.linear.x = lin
        msg.angular.z = ang
        self.pub.publish(msg)

def main(args=None):
    rclpy.init(args=args)
    rclpy.spin(CmdVelDriver())
    rclpy.shutdown()
Control: ros2 param set /cmd_vel_driver command forward → robot moves. ros2 param set /cmd_vel_driver command stop → stops.
